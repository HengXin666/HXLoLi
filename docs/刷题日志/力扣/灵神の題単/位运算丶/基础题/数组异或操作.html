1486. 数组异或操作

链接: 1486. 数组异或操作

给你两个整数，n 和 start 。

数组nums定义为：nums[i] = start + 2*i（下标从 0 开始）且 n == nums.length。

请返回nums中所有元素按位异或（XOR）后得到的结果。

题解

1. 模拟​

class Solution {
public:
    int xorOperation(int n, int start) {
        int res = 0;
        for (int i = 0; i < n; ++i) {
            res ^= start + (i * 2);
        }

        return res;
    }
};

class Solution {
public:
    int sumXor(int x) {
        if (x % 4 == 0) {
            return x;
        }
        if (x % 4 == 1) {
            return 1;
        }
        if (x % 4 == 2) {
            return x + 1;
        }
        return 0;
    }

    int xorOperation(int n, int start) {
        int s = start >> 1, e = n & start & 1;
        int ret = sumXor(s - 1) ^ sumXor(s + n - 1);
        return ret << 1 | e;
    }
};
