long long hxPow(long long a, long long b, long long mod) {

        b >>= 1;

    return res;

long long hxpow(long long x, long long p) {
    x %= mod;
    long long res = 1;
    while (p) {
        if (p & 1)
          res = (res * x) % mod;
        x = (x * x) % mod;
        p >>= 1;
    }
    return res;
}
