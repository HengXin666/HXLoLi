    using promise_type = Promise<T>;

    std::optional<NoVoidResType<T>> operator()() noexcept {
        if (_coroutine.done())
            return {};
        _coroutine.resume();

        constexpr auto await_suspend(std::coroutine_handle<> nowCoroutine) noexcept {
            _coroutine.promise()._mae = nowCoroutine;

};
