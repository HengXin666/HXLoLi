```cpp
constexpr int MAXX =  1e5;
vector<vector<int>> fac(MAXX + 10);

// 全局预处理每个数的质因数, fac[i][j] 表示 i 的质因数
auto __init__ = []{
    for (int i = 2; i <= MAXX; i++) 
        if (fac[i].empty()) 
            for (int j = i; j <= MAXX; j += i) 
                fac[j].push_back(i);
    return 0;
}();
```
